
-- This code stops cors from blocking requests. --

File Gemfile
	gem 'rack-cors', :require => 'rack/cors'

File config.ru
	require 'rack/cors'
	use Rack::Cors do

		# allow all origins in development
		allow do
			origins '*'
			resource '*',
					:headers => :any,
					:methods => [:get, :post, :delete, :put, :options]
		end
	end
